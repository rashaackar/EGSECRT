Google Sheets Integration
Connect to save every submitted vote automatically

Step 1 – Connect Your Sheet

Paste your Google Apps Script Web App URL below. Votes are sent on Final Submit.

Step 2 – Apps Script Code

Go to script.google.com → New Project → paste → Deploy as Web App (Anyone can access).

// ───────────────────────────────────────────────────── // Eden Global School — Assembly Election 2026-27 // Google Apps Script — Paste this ENTIRE code, // then Deploy → New Deployment → Web App → Anyone // ───────────────────────────────────────────────────── function doPost(e) { return saveRow(e.parameter); } function doGet(e) { return saveRow(e.parameter); } function saveRow(p) { try { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName("Votes"); if (!sheet) { sheet = ss.insertSheet("Votes"); } // Add header row if the sheet is empty if (sheet.getLastRow() === 0) { sheet.appendRow([ "Timestamp (IST)", "Voter ID", "Category", "Candidate", "Symbol", "Class" ]); var hdr = sheet.getRange(1, 1, 1, 6); hdr.setFontWeight("bold"); hdr.setBackground("#0a1628"); hdr.setFontColor("#c9a227"); hdr.setFontSize(11); sheet.setFrozenRows(1); sheet.setColumnWidth(1, 180); sheet.setColumnWidth(2, 140); sheet.setColumnWidth(3, 180); sheet.setColumnWidth(4, 180); sheet.setColumnWidth(5, 120); sheet.setColumnWidth(6, 80); } // Skip empty / test pings var cat = (p.category || "").trim(); if (cat === "" || cat === "TEST") { return ContentService .createTextOutput("SKIPPED") .setMimeType(ContentService.MimeType.TEXT); } sheet.appendRow([ new Date().toLocaleString("en-IN", { timeZone: "Asia/Kolkata" }), p.voterId || "", cat, p.candidate || "", p.symbol || "", p.cls || "" ]); return ContentService .createTextOutput("SAVED") .setMimeType(ContentService.MimeType.TEXT); } catch (err) { return ContentService .createTextOutput("ERROR: " + err.toString()) .setMimeType(ContentService.MimeType.TEXT); } }
